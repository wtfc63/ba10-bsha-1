00203         private double calculateDistance(TouchPoint a, double m1, double q1) {
00204                 // Orthogonal line m2*x + q2
00205                 double m2 = -1 / m1;
00206                 
00207                 // Calculate offset so the line goes through the point a
00208                 double q2 = a.y - m2 * a.x;
00209                 
00210                 // Calculate crossing point between the two lines
00211                 double x = (q2 - q1) / (m1 - m2);
