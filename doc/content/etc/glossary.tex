\makeglossaries


\newglossaryentry{glos:android}{
name={Android},
description={Linux-basiertes Betriebssystem für Mobiltelefone entwickelt von Google.}}

\newglossaryentry{glos:inputmethod}{
name={Input-Method},
description={Eine Input-Method ist der Teil des Android-Betriebssystems, der für die Benutzer-Eingabe von Text zuständig ist. Es ist möglich diese auszutauschen.}}

\newglossaryentry{Service}{name={service},
description={Ein Service für Android ist ein Programm im Hintergrund läuft und nicht direkt mit dem Benutzer in Kontakt kommt.}}
